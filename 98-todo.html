def _find_todo() -> Path:
    # walk up from the notebook working dir to the repo root
    for d in [Path.cwd(), *Path.cwd().parents]:
        if (d / "TODO.md").exists():
            return d / "TODO.md"
    raise FileNotFoundError("TODO.md not found above notebooks/")
